private class classname : parentClass {
    func int fibLinear(int a){
        int res;
        int fib = 0;
        int fibNext = 1;

        if(a == 0){
            res = a;
        }
        else{
            if(a == 1){
                res = 1;
            }
            else{
                int i = 1;

                loop(i < a){
                    int temp = fibNext;
                    fibNext = fib + fibNext;
                    fib = temp;
                    i = i + 1;
                }

                res = fibNext;
            }
        }

        return res;
    }

    func int fibRecursive(int a){
        int res;

        if(a == 0){
            res = 0;
        }else{
            if(a == 1){
                res = 1;
            }
            else{
                res = fibRecursive(a - 1) + fibRecursive(a - 2);
            }
        }
        return res;
    }

    main(){
        int a = 3;
        int x = fibRecursive(5);
        int y = fibLinear(7);
        print("Linear" + y);
        print("Recursive" + x);
    }
}
